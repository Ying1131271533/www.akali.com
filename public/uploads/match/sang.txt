	// code for fade in element by element
	$.fn.fadeInWithDelay = function(){
		var delay = 0;
		return this.each(function(){
			$(this).delay(delay).animate({opacity:1}, 200);
			delay += 100;
		});
	};

	function bindUpDownEvent()
	{
		$('.review-up').unbind('click');
		$('.review-down').unbind('click');
		$('.review-up').click(function(){
			if(!$(this).hasClass('disabled'))
			{
				var _this = this;
				$.get('/index.php?r=review/up&id='+$(this).attr('data-id'),function(rs){
					$("#up_"+$(_this).attr('data-id')).html(rs);
				})
			}
		});
		$('.review-down').click(function(){
			if(!$(this).hasClass('disabled'))
			{
				var _this = this;
				$.get('/index.php?r=review/down&id='+$(this).attr('data-id'),function(rs){
					$("#down_"+$(_this).attr('data-id')).html(rs);
				})
			}
		});
	}


	$(document).ready(function(){
		bindUpDownEvent();
		//查看更多评论信息
		if($('#content_wrapper').height()>160)
		{
			$('#view_more_summary').css('display','block').click(function(){
				$('#content_wrapper').parent().height('auto');
				$('#summary').height('auto');
				$(this).hide();
			});
		}

		//评分
		$('.star a').mouseenter(function(){
			var _parent = $(this).parent();
			if(_parent.hasClass('clicked')) return;
			_parent.removeClass('star2').removeClass('star4').removeClass('star6').removeClass('star8').removeClass('star10');
			_parent.addClass('star'+$(this).attr('data-score'));
		});
		$('.star').mouseout(function(){
			if($(this).hasClass('clicked')) return;
			$(this).removeClass('star2').removeClass('star4').removeClass('star6').removeClass('star8').removeClass('star10');
		});
		$('.star a').click(function(){
			if($(this).parent().hasClass('clicked')) return;
			$(this).parent().addClass('clicked');
			var vote = $(this).attr('data-score');
			$.get('/index.php?r=comic/vote&id=1420&vote='+vote,function(rs){
				if(rs!='')
				{
					obj = JSON.parse(rs);
					$('#vote_score').html(obj.data.score);
					$('#vote_count').html(obj.data.vote_count);
				}
			});

		});

		//加入收藏夹
		$("#add_to_favorite").click(function(){
			if(!$(this).hasClass('disabled'))
			{
				$(this).addClass('disabled');

			}
		});

		//自动加载评论
		var loadding = false;
		$('#review_list').scrollPagination({
			'contentPage': '/index.php?r=review/load',
			'contentData': {'comic_id':1420,'size':$('#review_list').children().size()},
			'scrollTarget': $(window),
			'heightOffset': 10,
			'beforeLoad': function(){

				if(14<20)
				{
					$('#review_list').stopScrollPagination();
					return;
				}
				$('#review_more').hide();
				$('#review_loading').fadeIn();	
			},
			'afterLoad': function(elementsLoaded){
				this.contentData.size = $('#review_list').children().size();
				bindUpDownEvent();
				$('#review_loading').fadeOut();
				$(elementsLoaded).fadeInWithDelay();
				if ($('#review_list').children().size() >= 100){ 
					$('#review_more').fadeIn();
					$('#review_list').stopScrollPagination();
				}
			}
		});

		//发表评论
		$("#btn_post").click(function(){
			var formdata = $('#review_form').serialize();
			if($.trim($('#content').val())=='')
			{
				layer.msg('内容不能为空！');
			}
			$.post("/index.php?r=review/save",formdata, function(data){
				data = JSON.parse(data);
				if(data.code==0)
				{
                	$("#content").val('');
                	//alert(data.message);
                	$.post('/index.php?r=review/load',{'comic_id':1420,'size':0},function(data){
                		$('#review_list').html('').append(data);
                		layer.msg('发表成功，谢谢参与！', {icon:9});
                		bindUpDownEvent();
                	});
            	}
			});
		});
